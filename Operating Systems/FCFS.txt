#include<stdio.h>
struct process
{ int btime,wtime,ttime,pid;}p[10];
int main()
{ int i,j,n;
  float ttut,twat;
  twat=0; ttut=0;
  printf("\nEnter the no' of processes: ");
  scanf("%d",&n);
  for(i=1;i<=n;i++)
  { printf("\nEnter the btime for process %d: ",i);
    scanf("%d",&p[i].btime);
    p[i].pid=i;}
  p[1].wtime=0;
  for(i=1;i<=n;i++)
  { p[i+1].wtime=p[i].wtime+p[i].btime;
    p[i].ttime=p[i].wtime+p[i].btime;
    ttut+=p[i].ttime;
    twat+=p[i].wtime;}
  printf("\n\nFCFS SCHEDULING\n\nProcess\tWtime\tBtime\tTtime");
  for(i=1;i<=n;i++)
  { printf("\n%d\t\t%d\t\t%d\t\t%d",p[i].pid,p[i].wtime,p[i].btime,p[i].ttime);}
  printf("\n\nAverage turn around time: %f\nAverage waiting time: %f",ttut/n,twat/n);
  printf("\n\nGNATT CHART:-\n");
  for(i=0;i<=50;i++)
  { printf("-");}
  printf("\n|");
  for(i=1;i<=n;i++)
  { for(j=0;j<p[i].btime/2;j++)
    { printf(" ");}
    printf("P%d",i);
    for(j=p[i].btime/2;j<p[i].btime;j++)
    { printf(" ");}
    printf("|");}
  printf("\n");
  for(i=0;i<=50;i++)
  { printf("-");}
  printf("\n0");
  for(i=1;i<=n;i++)
  { for(j=0;j<p[i].btime;j++)
    { printf(" ");}
    printf(" %d",p[i].ttime);}
  return 0;}
